# 箭頭函式中的this

let a = 0
let obj = {
    a: 'obj',
    foo() {
        console.log(this); // obj

        let test = ()=>{
            console.log(this); // obj
        }
        test()
    }
}
obj.foo()

    // => 箭頭函式內部沒有 this 指向, 箭頭函式的 this 指向是由外層函式的作用域決定的

# 能否更改箭頭函式的this指向?

function foo(){
    let test = ()=>{
        console.log(this);
    }
    return test
}

let obj1 = {
    a: 'obj1',
    foo
}

let obj2 = {
    a: 'obj2',
    foo
}

obj1.foo()() // 打印obj1
obj2.foo()() // 打印obj2

// => 默認綁定規則(純粹調用) 對箭頭函式無效, 因為箭頭函式里面沒有 this

# 案例一

var name = '我是window裡的name'
let obj1 = {
    name: 'obj1的name',
    fn1(){
        console.log(this.name);
    },
    fn2: ()=>{
        console.log(this.name);
    },
    fn3(){
        return function(){
            console.log(this.name);
        }
    },
    fn4(){
        return ()=>{
            console.log(this.name);
        }
    }
}
let obj2 = {
    name : 'obj2的name'
}

obj1.fn1(); // ?
obj1.fn1.call(obj2); // ?

obj1.fn2(); // ?
obj1.fn2.call(obj2); // ?

obj1.fn3()(); // ?
obj1.fn3().call(obj2); // ?
obj1.fn3.call(obj2)(); // ?

obj1.fn4()() // ?
obj1.fn4().call(obj2); // ?
obj1.fn4.call(obj2)(); // ?
